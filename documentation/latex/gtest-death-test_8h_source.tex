\DoxyCodeLine{69 \textcolor{comment}{// Here's what happens when an ASSERT\_DEATH* or EXPECT\_DEATH* is}}
\DoxyCodeLine{70 \textcolor{comment}{// executed:}}
\DoxyCodeLine{71 \textcolor{comment}{//}}
\DoxyCodeLine{72 \textcolor{comment}{//   1. It generates a warning if there is more than one active}}
\DoxyCodeLine{73 \textcolor{comment}{//   thread.  This is because it's safe to fork() or clone() only}}
\DoxyCodeLine{74 \textcolor{comment}{//   when there is a single thread.}}
\DoxyCodeLine{75 \textcolor{comment}{//}}
\DoxyCodeLine{76 \textcolor{comment}{//   2. The parent process clone()s a sub-\/process and runs the death}}
\DoxyCodeLine{77 \textcolor{comment}{//   test in it; the sub-\/process exits with code 0 at the end of the}}
\DoxyCodeLine{78 \textcolor{comment}{//   death test, if it hasn't exited already.}}
\DoxyCodeLine{79 \textcolor{comment}{//}}
\DoxyCodeLine{80 \textcolor{comment}{//   3. The parent process waits for the sub-\/process to terminate.}}
\DoxyCodeLine{81 \textcolor{comment}{//}}
\DoxyCodeLine{82 \textcolor{comment}{//   4. The parent process checks the exit code and error message of}}
\DoxyCodeLine{83 \textcolor{comment}{//   the sub-\/process.}}
\DoxyCodeLine{84 \textcolor{comment}{//}}
\DoxyCodeLine{85 \textcolor{comment}{// Examples:}}
\DoxyCodeLine{86 \textcolor{comment}{//}}
\DoxyCodeLine{87 \textcolor{comment}{//   ASSERT\_DEATH(server.SendMessage(56, "{}Hello"{}), "{}Invalid port number"{});}}
\DoxyCodeLine{88 \textcolor{comment}{//   for (int i = 0; i < 5; i++) \{}}
\DoxyCodeLine{89 \textcolor{comment}{//     EXPECT\_DEATH(server.ProcessRequest(i),}}
\DoxyCodeLine{90 \textcolor{comment}{//                  "{}Invalid request .* in ProcessRequest()"{})}}
\DoxyCodeLine{91 \textcolor{comment}{//                  << "{}Failed to die on request "{} << i;}}
\DoxyCodeLine{92 \textcolor{comment}{//   \}}}
\DoxyCodeLine{93 \textcolor{comment}{//}}
\DoxyCodeLine{94 \textcolor{comment}{//   ASSERT\_EXIT(server.ExitNow(), ::testing::ExitedWithCode(0), "{}Exiting"{});}}
\DoxyCodeLine{95 \textcolor{comment}{//}}
\DoxyCodeLine{96 \textcolor{comment}{//   bool KilledBySIGHUP(int exit\_code) \{}}
\DoxyCodeLine{97 \textcolor{comment}{//     return WIFSIGNALED(exit\_code) \&\& WTERMSIG(exit\_code) == SIGHUP;}}
\DoxyCodeLine{98 \textcolor{comment}{//   \}}}
\DoxyCodeLine{99 \textcolor{comment}{//}}
\DoxyCodeLine{100 \textcolor{comment}{//   ASSERT\_EXIT(client.HangUpServer(), KilledBySIGHUP, "{}Hanging up!"{});}}
\DoxyCodeLine{101 \textcolor{comment}{//}}
\DoxyCodeLine{102 \textcolor{comment}{// The final parameter to each of these macros is a matcher applied to any data}}
\DoxyCodeLine{103 \textcolor{comment}{// the sub-\/process wrote to stderr.  For compatibility with existing tests, a}}
\DoxyCodeLine{104 \textcolor{comment}{// bare string is interpreted as a regular expression matcher.}}
\DoxyCodeLine{105 \textcolor{comment}{//}}
\DoxyCodeLine{106 \textcolor{comment}{// On the regular expressions used in death tests:}}
\DoxyCodeLine{107 \textcolor{comment}{//}}
\DoxyCodeLine{108 \textcolor{comment}{//   On POSIX-\/compliant systems (*nix), we use the <regex.h> library,}}
\DoxyCodeLine{109 \textcolor{comment}{//   which uses the POSIX extended regex syntax.}}
\DoxyCodeLine{110 \textcolor{comment}{//}}
\DoxyCodeLine{111 \textcolor{comment}{//   On other platforms (e.g. Windows or Mac), we only support a simple regex}}
\DoxyCodeLine{112 \textcolor{comment}{//   syntax implemented as part of Google Test.  This limited}}
\DoxyCodeLine{113 \textcolor{comment}{//   implementation should be enough most of the time when writing}}
\DoxyCodeLine{114 \textcolor{comment}{//   death tests; though it lacks many features you can find in PCRE}}
\DoxyCodeLine{115 \textcolor{comment}{//   or POSIX extended regex syntax.  For example, we don't support}}
\DoxyCodeLine{116 \textcolor{comment}{//   union ("{}x|y"{}), grouping ("{}(xy)"{}), brackets ("{}[xy]"{}), and}}
\DoxyCodeLine{117 \textcolor{comment}{//   repetition count ("{}x\{5,7\}"{}), among others.}}
\DoxyCodeLine{118 \textcolor{comment}{//}}
\DoxyCodeLine{119 \textcolor{comment}{//   Below is the syntax that we do support.  We chose it to be a}}
\DoxyCodeLine{120 \textcolor{comment}{//   subset of both PCRE and POSIX extended regex, so it's easy to}}
\DoxyCodeLine{121 \textcolor{comment}{//   learn wherever you come from.  In the following: 'A' denotes a}}
\DoxyCodeLine{122 \textcolor{comment}{//   literal character, period (.), or a single \(\backslash\)\(\backslash\) escape sequence;}}
\DoxyCodeLine{123 \textcolor{comment}{//   'x' and 'y' denote regular expressions; 'm' and 'n' are for}}
\DoxyCodeLine{124 \textcolor{comment}{//   natural numbers.}}
\DoxyCodeLine{125 \textcolor{comment}{//}}
\DoxyCodeLine{126 \textcolor{comment}{//     c     matches any literal character c}}
\DoxyCodeLine{127 \textcolor{comment}{//     \(\backslash\)\(\backslash\)d   matches any decimal digit}}
\DoxyCodeLine{128 \textcolor{comment}{//     \(\backslash\)\(\backslash\)D   matches any character that's not a decimal digit}}
\DoxyCodeLine{129 \textcolor{comment}{//     \(\backslash\)\(\backslash\)f   matches \(\backslash\)f}}
\DoxyCodeLine{130 \textcolor{comment}{//     \(\backslash\)\(\backslash\)n   matches \(\backslash\)n}}
\DoxyCodeLine{131 \textcolor{comment}{//     \(\backslash\)\(\backslash\)r   matches \(\backslash\)r}}
\DoxyCodeLine{132 \textcolor{comment}{//     \(\backslash\)\(\backslash\)s   matches any ASCII whitespace, including \(\backslash\)n}}
\DoxyCodeLine{133 \textcolor{comment}{//     \(\backslash\)\(\backslash\)S   matches any character that's not a whitespace}}
\DoxyCodeLine{134 \textcolor{comment}{//     \(\backslash\)\(\backslash\)t   matches \(\backslash\)t}}
\DoxyCodeLine{135 \textcolor{comment}{//     \(\backslash\)\(\backslash\)v   matches \(\backslash\)v}}
\DoxyCodeLine{136 \textcolor{comment}{//     \(\backslash\)\(\backslash\)w   matches any letter, \_, or decimal digit}}
\DoxyCodeLine{137 \textcolor{comment}{//     \(\backslash\)\(\backslash\)W   matches any character that \(\backslash\)\(\backslash\)w doesn't match}}
\DoxyCodeLine{138 \textcolor{comment}{//     \(\backslash\)\(\backslash\)c   matches any literal character c, which must be a punctuation}}
\DoxyCodeLine{139 \textcolor{comment}{//     .     matches any single character except \(\backslash\)n}}
\DoxyCodeLine{140 \textcolor{comment}{//     A?    matches 0 or 1 occurrences of A}}
\DoxyCodeLine{141 \textcolor{comment}{//     A*    matches 0 or many occurrences of A}}
\DoxyCodeLine{142 \textcolor{comment}{//     A+    matches 1 or many occurrences of A}}
\DoxyCodeLine{143 \textcolor{comment}{//     \string^     matches the beginning of a string (not that of each line)}}
\DoxyCodeLine{144 \textcolor{comment}{//     \$     matches the end of a string (not that of each line)}}
\DoxyCodeLine{145 \textcolor{comment}{//     xy    matches x followed by y}}
\DoxyCodeLine{146 \textcolor{comment}{//}}
\DoxyCodeLine{147 \textcolor{comment}{//   If you accidentally use PCRE or POSIX extended regex features}}
\DoxyCodeLine{148 \textcolor{comment}{//   not implemented by us, you will get a run-\/time failure.  In that}}
\DoxyCodeLine{149 \textcolor{comment}{//   case, please try to rewrite your regular expression within the}}
\DoxyCodeLine{150 \textcolor{comment}{//   above syntax.}}
\DoxyCodeLine{151 \textcolor{comment}{//}}
\DoxyCodeLine{152 \textcolor{comment}{//   This implementation is *not* meant to be as highly tuned or robust}}
\DoxyCodeLine{153 \textcolor{comment}{//   as a compiled regex library, but should perform well enough for a}}
\DoxyCodeLine{154 \textcolor{comment}{//   death test, which already incurs significant overhead by launching}}
\DoxyCodeLine{155 \textcolor{comment}{//   a child process.}}
\DoxyCodeLine{156 \textcolor{comment}{//}}
\DoxyCodeLine{157 \textcolor{comment}{// Known caveats:}}
\DoxyCodeLine{158 \textcolor{comment}{//}}
\DoxyCodeLine{159 \textcolor{comment}{//   A "{}threadsafe"{} style death test obtains the path to the test}}
\DoxyCodeLine{160 \textcolor{comment}{//   program from argv[0] and re-\/executes it in the sub-\/process.  For}}
\DoxyCodeLine{161 \textcolor{comment}{//   simplicity, the current implementation doesn't search the PATH}}
\DoxyCodeLine{162 \textcolor{comment}{//   when launching the sub-\/process.  This means that the user must}}
\DoxyCodeLine{163 \textcolor{comment}{//   invoke the test program via a path that contains at least one}}
\DoxyCodeLine{164 \textcolor{comment}{//   path separator (e.g. path/to/foo\_test and}}
\DoxyCodeLine{165 \textcolor{comment}{//   /absolute/path/to/bar\_test are fine, but foo\_test is not).  This}}
\DoxyCodeLine{166 \textcolor{comment}{//   is rarely a problem as people usually don't put the test binary}}
\DoxyCodeLine{167 \textcolor{comment}{//   directory in PATH.}}
\DoxyCodeLine{168 \textcolor{comment}{//}}
