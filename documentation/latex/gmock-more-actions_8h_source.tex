\DoxyCodeLine{49 \textcolor{comment}{// Sometimes you want to give an action explicit template parameters}}
\DoxyCodeLine{50 \textcolor{comment}{// that cannot be inferred from its value parameters.  ACTION() and}}
\DoxyCodeLine{51 \textcolor{comment}{// ACTION\_P*() don't support that.  ACTION\_TEMPLATE() remedies that}}
\DoxyCodeLine{52 \textcolor{comment}{// and can be viewed as an extension to ACTION() and ACTION\_P*().}}
\DoxyCodeLine{53 \textcolor{comment}{//}}
\DoxyCodeLine{54 \textcolor{comment}{// The syntax:}}
\DoxyCodeLine{55 \textcolor{comment}{//}}
\DoxyCodeLine{56 \textcolor{comment}{//   ACTION\_TEMPLATE(ActionName,}}
\DoxyCodeLine{57 \textcolor{comment}{//                   HAS\_m\_TEMPLATE\_PARAMS(kind1, name1, ..., kind\_m, name\_m),}}
\DoxyCodeLine{58 \textcolor{comment}{//                   AND\_n\_VALUE\_PARAMS(p1, ..., p\_n)) \{ statements; \}}}
\DoxyCodeLine{59 \textcolor{comment}{//}}
\DoxyCodeLine{60 \textcolor{comment}{// defines an action template that takes m explicit template}}
\DoxyCodeLine{61 \textcolor{comment}{// parameters and n value parameters.  name\_i is the name of the i-\/th}}
\DoxyCodeLine{62 \textcolor{comment}{// template parameter, and kind\_i specifies whether it's a typename,}}
\DoxyCodeLine{63 \textcolor{comment}{// an integral constant, or a template.  p\_i is the name of the i-\/th}}
\DoxyCodeLine{64 \textcolor{comment}{// value parameter.}}
\DoxyCodeLine{65 \textcolor{comment}{//}}
\DoxyCodeLine{66 \textcolor{comment}{// Example:}}
\DoxyCodeLine{67 \textcolor{comment}{//}}
\DoxyCodeLine{68 \textcolor{comment}{//   // DuplicateArg<k, T>(output) converts the k-\/th argument of the mock}}
\DoxyCodeLine{69 \textcolor{comment}{//   // function to type T and copies it to *output.}}
\DoxyCodeLine{70 \textcolor{comment}{//   ACTION\_TEMPLATE(DuplicateArg,}}
\DoxyCodeLine{71 \textcolor{comment}{//                   HAS\_2\_TEMPLATE\_PARAMS(int, k, typename, T),}}
\DoxyCodeLine{72 \textcolor{comment}{//                   AND\_1\_VALUE\_PARAMS(output)) \{}}
\DoxyCodeLine{73 \textcolor{comment}{//     *output = T(::std::get<k>(args));}}
\DoxyCodeLine{74 \textcolor{comment}{//   \}}}
\DoxyCodeLine{75 \textcolor{comment}{//   ...}}
\DoxyCodeLine{76 \textcolor{comment}{//     int n;}}
\DoxyCodeLine{77 \textcolor{comment}{//     EXPECT\_CALL(mock, Foo(\_, \_))}}
\DoxyCodeLine{78 \textcolor{comment}{//         .WillOnce(DuplicateArg<1, unsigned char>(\&n));}}
\DoxyCodeLine{79 \textcolor{comment}{//}}
\DoxyCodeLine{80 \textcolor{comment}{// To create an instance of an action template, write:}}
\DoxyCodeLine{81 \textcolor{comment}{//}}
\DoxyCodeLine{82 \textcolor{comment}{//   ActionName<t1, ..., t\_m>(v1, ..., v\_n)}}
\DoxyCodeLine{83 \textcolor{comment}{//}}
\DoxyCodeLine{84 \textcolor{comment}{// where the ts are the template arguments and the vs are the value}}
\DoxyCodeLine{85 \textcolor{comment}{// arguments.  The value argument types are inferred by the compiler.}}
\DoxyCodeLine{86 \textcolor{comment}{// If you want to explicitly specify the value argument types, you can}}
\DoxyCodeLine{87 \textcolor{comment}{// provide additional template arguments:}}
\DoxyCodeLine{88 \textcolor{comment}{//}}
\DoxyCodeLine{89 \textcolor{comment}{//   ActionName<t1, ..., t\_m, u1, ..., u\_k>(v1, ..., v\_n)}}
\DoxyCodeLine{90 \textcolor{comment}{//}}
\DoxyCodeLine{91 \textcolor{comment}{// where u\_i is the desired type of v\_i.}}
\DoxyCodeLine{92 \textcolor{comment}{//}}
\DoxyCodeLine{93 \textcolor{comment}{// ACTION\_TEMPLATE and ACTION/ACTION\_P* can be overloaded on the}}
\DoxyCodeLine{94 \textcolor{comment}{// number of value parameters, but not on the number of template}}
\DoxyCodeLine{95 \textcolor{comment}{// parameters.  Without the restriction, the meaning of the following}}
\DoxyCodeLine{96 \textcolor{comment}{// is unclear:}}
\DoxyCodeLine{97 \textcolor{comment}{//}}
\DoxyCodeLine{98 \textcolor{comment}{//   OverloadedAction<int, bool>(x);}}
\DoxyCodeLine{99 \textcolor{comment}{//}}
\DoxyCodeLine{100 \textcolor{comment}{// Are we using a single-\/template-\/parameter action where 'bool' refers}}
\DoxyCodeLine{101 \textcolor{comment}{// to the type of x, or are we using a two-\/template-\/parameter action}}
\DoxyCodeLine{102 \textcolor{comment}{// where the compiler is asked to infer the type of x?}}
\DoxyCodeLine{103 \textcolor{comment}{//}}
\DoxyCodeLine{104 \textcolor{comment}{// Implementation notes:}}
\DoxyCodeLine{105 \textcolor{comment}{//}}
\DoxyCodeLine{106 \textcolor{comment}{// GMOCK\_INTERNAL\_*\_HAS\_m\_TEMPLATE\_PARAMS and}}
\DoxyCodeLine{107 \textcolor{comment}{// GMOCK\_INTERNAL\_*\_AND\_n\_VALUE\_PARAMS are internal macros for}}
\DoxyCodeLine{108 \textcolor{comment}{// implementing ACTION\_TEMPLATE.  The main trick we use is to create}}
\DoxyCodeLine{109 \textcolor{comment}{// new macro invocations when expanding a macro.  For example, we have}}
\DoxyCodeLine{110 \textcolor{comment}{//}}
\DoxyCodeLine{111 \textcolor{comment}{//   \#define ACTION\_TEMPLATE(name, template\_params, value\_params)}}
\DoxyCodeLine{112 \textcolor{comment}{//       ... GMOCK\_INTERNAL\_DECL\_\#\#template\_params ...}}
\DoxyCodeLine{113 \textcolor{comment}{//}}
\DoxyCodeLine{114 \textcolor{comment}{// which causes ACTION\_TEMPLATE(..., HAS\_1\_TEMPLATE\_PARAMS(typename, T), ...)}}
\DoxyCodeLine{115 \textcolor{comment}{// to expand to}}
\DoxyCodeLine{116 \textcolor{comment}{//}}
\DoxyCodeLine{117 \textcolor{comment}{//       ... GMOCK\_INTERNAL\_DECL\_HAS\_1\_TEMPLATE\_PARAMS(typename, T) ...}}
\DoxyCodeLine{118 \textcolor{comment}{//}}
\DoxyCodeLine{119 \textcolor{comment}{// Since GMOCK\_INTERNAL\_DECL\_HAS\_1\_TEMPLATE\_PARAMS is a macro, the}}
\DoxyCodeLine{120 \textcolor{comment}{// preprocessor will continue to expand it to}}
\DoxyCodeLine{121 \textcolor{comment}{//}}
\DoxyCodeLine{122 \textcolor{comment}{//       ... typename T ...}}
\DoxyCodeLine{123 \textcolor{comment}{//}}
\DoxyCodeLine{124 \textcolor{comment}{// This technique conforms to the C++ standard and is portable.  It}}
\DoxyCodeLine{125 \textcolor{comment}{// allows us to implement action templates using O(N) code, where N is}}
\DoxyCodeLine{126 \textcolor{comment}{// the maximum number of template/value parameters supported.  Without}}
\DoxyCodeLine{127 \textcolor{comment}{// using it, we'd have to devote O(N\string^2) amount of code to implement all}}
\DoxyCodeLine{128 \textcolor{comment}{// combinations of m and n.}}
\DoxyCodeLine{129 }
