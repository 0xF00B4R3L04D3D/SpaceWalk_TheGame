467 void loadRooms(tinyxml2::XMLElement* firstRoom) {

469 tinyxml2::XMLElement* actual = firstRoom;

472 const std::string roomDescription(actual->FirstChildElement("description")->GetText());
473 int roomID = 0;
474 actual->FirstChildElement("id")->QueryIntText(&roomID);
475 RoomFactory(roomName, roomID, roomDescription); // Construct room with name, id and desc
476 tinyxml2::XMLElement* invEle = actual->FirstChildElement("inventory");
477 tinyxml2::XMLElement* actualInv = invEle->FirstChildElement("object");
478 while(actualInv) {
479 const std::string objName(actualInv->FirstChildElement("name")->GetText());
480 const std::string desc(actualInv->FirstChildElement("description")->GetText());
481 int objectID = 0;

483 auto tmpItem = std::make_unique<Object>(objName, objectID, desc);
484 worldRooms.back()->addItem(tmpItem);
485 actualInv = actualInv->NextSiblingElement("object");
486 }
487 tinyxml2::XMLElement* conns = actual->FirstChildElement("connections");
488 roomConnectionMap.insert(std::make_pair(roomID, trackConnections(conns)));
489 // While loading entities into population, the number of entities loaded must be returned to be able to add these enitites to the room they should be in.
490 int entCount = loadEntities(actual->FirstChildElement("entity"));
491 for (int i = int(population.size())-entCount; i < int(population.size()); i++) {
492 worldRooms.back()->addEntity(population[i]);
493 }
494 actual = actual->NextSiblingElement(elementName);
495 }

501 void connectRooms() {
502 for (std::map<int, std::vector<int>>::iterator it = roomConnectionMap.begin(); it != roomConnectionMap.end(); it++) {
503 int parentID = it->first;
504 nodes::iterator parentRoom = std::find_if(worldRooms.begin(), worldRooms.end(),
505 [&parentID](node& r) {if (r->getID() == parentID) return true; return false;});
506 if (parentRoom != worldRooms.end()) {
507 for (std::vector<int>::iterator sit = it->second.begin(); sit != it->second.end(); sit++) {
508 int id2find = *sit;
509 nodes::iterator childRoom = std::find_if(worldRooms.begin(), worldRooms.end(),
510 [&id2find](node& r) {if (r->getID() == id2find) return true; return false;});
511 if (childRoom != worldRooms.end()) {
512 parentRoom->get()->addNeighbour(*childRoom);
513 }
514 }
515 }
516 }
